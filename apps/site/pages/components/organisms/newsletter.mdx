import { Tab, Tabs } from 'nextra-theme-docs'
import PropsSection from 'site/components/PropsSection'
import { TokenTable, TokenRow, TokenDivider } from 'site/components/Tokens'
import { OverviewSection } from 'site/components/OverviewSection'
import { SectionItem, SectionList } from 'site/components/SectionItem'
import { Newsletter, Icon } from '@faststore/ui'

import path from 'path'
import { useSSG } from 'nextra/ssg'
import { getComponentPropsFrom } from 'site/components/utilities/propsSection'

export const getStaticProps = () => {
  const newsletterPath = path.resolve(__filename)
  const components = ['Newsletter.tsx']
  const [newsletterProps] = getComponentPropsFrom(newsletterPath, components)
  return {
    props: {
      // We add an `ssg` field to the page props,
      // which will be provided to the Nextra `useSSG` hook.
      ssg: {
        newsletterProps,
      },
    },
  }
}

export const NewsletterPropsSection = ({ component }) => {
  // Get the data from SSG, and render it as a component.
  const { newsletterProps } = useSSG()
  return {
    Newsletter: <PropsSection propsList={newsletterProps} />,
  }[component]
}

<header>

# Newsletter

The Newsletter component is a section with at least one input text with an action which its value could be submitted.

</header>

## Import

Import the component from [@faststore/ui](/../components)

```tsx
import { Newsletter } from '@faststore/ui'
```

Import Styles

```tsx
import '@faststore/ui/src/components/organisms/Newsletter/styles.scss'
```

## Props

All newsletter related components support all attributes also supported by the `<div>` tag.
Besides those attributes, the following props are also supported:

<NewsletterPropsSection component="Newsletter" />
